    let currentTab = 'whatsapp';

 

    function updatePreview() {

        const lang = document.getElementById('language').value;

        let amount = parseInt(document.getElementById('minInvest').value) || 2500;

       

        // Enforce the legal minimal constraint on UI layer

        if (amount < 2500) {

            amount = 2500;

            document.getElementById('minInvest').value = 2500;

        }

 

        const name = document.getElementById('partnerName').value || '[Partner Name]';

        const mobile = document.getElementById('partnerMob').value || '[Mobile Number]';

        const arn = document.getElementById('partnerArn').value || '[ARN Number]';

 

        let template = contentMatrix[lang][currentTab] || contentMatrix['en'][currentTab];

       

        // Template token evaluation execution

        let processed = template

            .replace(/{{AMOUNT}}/g, amount.toLocaleString('en-IN'))

            .replace(/{{NAME}}/g, name)

            .replace(/{{MOBILE}}/g, mobile)

            .replace(/{{ARN}}/g, arn);

 

        document.getElementById('previewArea').textContent = processed;

    }

 

    function switchTab(tabName) {

        document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));

        event.target.classList.add('active');

        currentTab = tabName;

        updatePreview();

    }

 

    function triggerDownload() {

        const textData = document.getElementById('previewArea').textContent;

        const blob = new Blob([textData], { type: 'text/plain;charset=utf-8' });

        const link = document.createElement('a');

        link.href = URL.createObjectURL(blob);

        link.download = `HighGrowth_Basket_${currentTab}_${document.getElementById('language').value}.txt`;

        link.click();
